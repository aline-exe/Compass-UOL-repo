

select
    autor.codautor as codautor,
    autor.nome as nome,
    count (*) as quantidade_publicacoes
from
    autor

left join
    livro on autor.codautor = livro.autor

group by
    autor.codautor, autor.nascimento, autor.nome

order by
   quantidade_publicacoes desc
limit 1

-- codautor, nome, quantidade_publicacoes.
